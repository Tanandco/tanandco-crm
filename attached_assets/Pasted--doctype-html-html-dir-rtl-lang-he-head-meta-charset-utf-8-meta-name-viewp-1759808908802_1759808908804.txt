    // Splash — סגירה אוטומטית אחרי 2.4ש׳ או בלחיצה
    (function(){
      const splash = document.getElementById('splash');
      if(!splash) return;
      let closed = false;
      const close = () => { if(closed) return; closed = true; splash.style.animation = 'fadeOut .6s ease forwards'; };

    (function(){
      const vids = Array.from(document.querySelectorAll('video'));
      const io = new IntersectionObserver((entries)=>{
        entries.forEach(e=>{
          if(e.isIntersecting){ e.target.play().catch(()=>{}); }
          else { e.target.pause(); }
        });
      }, { threshold: 0.25 });
      vids.forEach(v=>{ try{ v.play(); }catch{} io.observe(v); });
    })();
